#include <amxmodx>
#include <amxmisc>

#define PLUGIN "_settings"
#define VERSION "1.0"
#define AUTHOR "1xAero"

public plugin_init()
	register_plugin(PLUGIN, VERSION, AUTHOR);

public plugin_cfg()
{
	//Get cfgdir.
	new cfgdir[32];
	get_configsdir(cfgdir, charsmax(cfgdir));
	
	//Execute.
	server_cmd("exec %s/%s", cfgdir, "aerobot.cfg");
}
