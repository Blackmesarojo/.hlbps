#include <amxmodx>
#include <hamsandwich>

#define PLUGIN  "Func_Vehicle for admins only"
#define VERSION "1.0"
#define AUTHOR  "Sneaky.amxx"

public plugin_init()
{
	register_plugin(PLUGIN, VERSION, AUTHOR)
	RegisterHam(Ham_Use, "func_vehicle", "CheckPriv");
    //RegisterHam(Ham_Use, "func_vehiclecontrol", "CheckPriv");	
}
public CheckPriv(ent, id, activator, iType, Float:fValue)
{
    if( get_user_flags(id) & ADMIN_KICK )
    {
        return HAM_IGNORED;
    }
    else if(id)
    {
    	client_print(id, print_center, "Only VIPs Or Admins Can Drive :) Type /buyvip");
    	return HAM_SUPERCEDE
    }
    return HAM_IGNORED;
}
